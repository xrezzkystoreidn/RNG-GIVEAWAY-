🤖 Bot Integration Docs

Update ke bot kamu dengan path RTDB berikut. Project: accdeposit-422a2

📁 Struktur RTDB

/users/{phone}/           ← data user (role, saldo, nama)
/deposits/{pushKey}/      ← request deposit
/orders/{pushKey}/        ← transaksi order
/products/{pushKey}/      ← katalog produk (dikelola admin panel)
/config/site/             ← setting toko
/config/saldootp/         ← saldo OTP provider
/otp_logs/{pushKey}/      ← log OTP terkirim

👤 1. User Daftar → set ke /users/{phone}

// WAJIB set field ini persis:
await db.ref(`users/${phone}`).set({
  uid: phone, phone, name, pushName,
  role: 'user',        // JANGAN 'admin'
  saldo: 0, totalDeposit: 0, totalOrder: 0,
  isActive: true, isBanned: false,
  registeredFrom: 'whatsapp-bot',
  createdAt: Date.now()
});

💰 2. Deposit Request → push ke /deposits/ + cek spam

// WAJIB cek pending dulu — anti-spam
const existing = await db.ref('deposits')
  .orderByChild('phone').equalTo(phone).get();
let hasPending = false;
existing.forEach(c => {
  if (c.val().status === 'pending') hasPending = true;
});
if (hasPending) {
  return bot.reply('❌ Kamu masih punya deposit pending!');
}
// Baru buat deposit
await db.ref('deposits').push({
  phone, name, amount, method: 'Transfer Bank',
  status: 'pending', hasBukti: false, buktiUrl: null,
  createdAt: Date.now()
});

📦 3. Order / Beli Produk → auto kurangi saldo

// Ambil produk dari RTDB (dikelola admin panel)
const prodSnap = await db.ref(`products/${prodKey}`).get();
const prod = prodSnap.val();

// Cek user & saldo
const userSnap = await db.ref(`users/${phone}`).get();
const user = userSnap.val();
if (user.isBanned) return bot.reply('❌ Akun dibanned.');
if (user.saldo < prod.harga) return bot.reply('❌ Saldo kurang.');

// Buat order + kurangi saldo (atomic transaction)
const ordRef = db.ref('orders').push();
await ordRef.set({
  phone, userId: phone, name: user.name,
  prodKey, service: prod.name, kategori: prod.kategori,
  harga: prod.harga, target: targetId,
  status: 'pending', createdAt: Date.now()
});
// Kurangi saldo pakai transaction (atomic, aman)
await db.ref(`users/${phone}/saldo`)
  .transaction(s => Math.max(0, (s||0) - prod.harga));
await db.ref(`users/${phone}/totalOrder`)
  .transaction(t => (t||0) + 1);
// Kurangi stok kalau bukan unlimited
if (prod.stok !== -1) {
  await db.ref(`products/${prodKey}/stok`)
    .transaction(s => Math.max(0, (s||0) - 1));
}

🔔 4. Listen Notif Approve/Reject (Bot harus listen ini)

db.ref('deposits').on('child_changed', async snap => {
  const d = snap.val();
  if (d.notifApproved === true) {
    await bot.sendMessage(d.phone, d.notifMsg);
    await snap.ref.update({ notifApproved: false });
  }
  if (d.notifRejected === true) {
    await bot.sendMessage(d.phone,
      `❌ Deposit Rp ${d.amount} ditolak. Alasan: ${d.alasan||'-'}`);
    await snap.ref.update({ notifRejected: false });
  }
});
// Listen status order berubah:
db.ref('orders').on('child_changed', async snap => {
  const o = snap.val();
  if (o.notifStatus && !o.notifSent) {
    await bot.sendMessage(o.phone, o.notifStatus);
    await snap.ref.update({ notifSent: true });
  }
});
